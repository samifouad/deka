<Function name="count">
  <Description>
    Counts the number of elements in an array. Null returns 0, non-array values
    return 1. Pass mode 1 to count recursively.
  </Description>
  <Parameter name="$value" type="mixed" required="true" typeLink="/docs/php/types/mixed">
    The value to count.
  </Parameter>
  <Parameter name="$mode" type="int" required="false" typeLink="/docs/php/types/int">
    When 1, counts nested arrays recursively.
  </Parameter>
  <ReturnType type="int" typeLink="/docs/php/types/int" />
</Function>
