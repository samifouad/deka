<Function name="get_str_func">
  <Description>
    Returns a string representation of a PHP value. Use this helper when the
    runtime needs a safe string form for logging or error messages without
    mutating the value itself.
  </Description>
  <Parameter name="$value" type="mixed" typeLink="/docs/php/types/mixed">
    The value to stringify.
  </Parameter>
  <ReturnType type="string" typeLink="/docs/php/types/string" />
</Function>

## Examples

```php
<?php

$value = ["status" => "ok", "count" => 3];
$stringValue = get_str_func($value);

echo $stringValue;

?>
```
