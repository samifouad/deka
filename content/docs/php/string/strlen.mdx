<Function name="strlen">
  <Description>
    Returns the length of a string in bytes. For ASCII, bytes equal characters.
    Use [mb_strlen()](/docs/php/string/mb_strlen) for multibyte character counts.
    Arrays and objects return null.
  </Description>
  <Parameter name="$string" type="string" required="true" typeLink="/docs/php/types/string">
    The string to measure.
  </Parameter>
  <ReturnType type="int" typeLink="/docs/php/types/int" />
</Function>
