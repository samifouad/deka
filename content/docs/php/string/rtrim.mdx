<Function name="rtrim">
  <Description>
    Strips whitespace (or other characters) from the end of a string.
    Defaults to trimming " \n\r\t\v\x00" (space, tab, newline, carriage return,
    NUL, vertical tab).
  </Description>
  <Parameter name="$string" type="string" required="true" typeLink="/docs/php/types/string">
    The string to trim.
  </Parameter>
  <Parameter name="$characters" type="string" required="false" typeLink="/docs/php/types/string">
    Optional character mask to strip from the end.
  </Parameter>
  <ReturnType type="string" typeLink="/docs/php/types/string" />
</Function>
